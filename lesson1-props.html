درس Props في React: تحويل المكونات الثابتة إلى مكونات قابلة لإعادة الاستخدام

المرحلة الأولى: المكونات الثابتة (بدون Props)

في هذه المرحلة، نقوم بإنشاء مكون منفصل لكل مشروع. هذه الطريقة غير فعالة لأنها:

  • تحتاج إلى تكرار الكود لكل مشروع جديد
  • تصعب الصيانة عند الحاجة لتغيير هيكل المكون
  • تخالف مبدأ DRY (Don't Repeat Yourself)
// مكونات المشاريع المنفصلة - الطريقة غير المثالية
function Project1() {
    return (
        <div className="project">
            <h1>Admin Dashboard</h1>
            <p>لوحة تحكم إدارية...</p>
        </div>
    );
}

function Project2() {
    return (
        <div className="project">
            <h1>E-commerce Platform</h1>
            <p>منصة تجارة إلكترونية...</p>
        </div>
    );
}

function Project3() {
    return (
        <div className="project">
            <h1>Social Media App</h1>
            <p>تطبيق تواصل اجتماعي...</p>
        </div>
    );
}

// المكون الرئيسي للتطبيق
function App() {
    return (
        <div>
            <h1 className="page-title">مشاريعي</h1>
            <Project1 />
            <Project2 />
            <Project3 />
        </div>
    );
}

// تحميل التطبيق
const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
المشكلة في هذا الكود:
  • تكرار نفس الهيكل البرمجي ثلاث مرات
  • صعوبة الصيانة عند الحاجة لتغيير شكل المكونات
  • الحاجة لإنشاء مكون جديد لكل مشروع جديد

المرحلة الثانية: استخدام Props

بدلاً من إنشاء مكون منفصل لكل مشروع، يمكننا إنشاء مكون واحد يستقبل البيانات كـ props:

// مكون المشروع القابل لإعادة الاستخدام
function Project(props) {
    return (
        <div className="project">
            <h1>{props.title}</h1>
            <p>{props.description}</p>
        </div>
    );
}

// المكون الرئيسي للتطبيق
function App() {
    // تخزين بيانات المشاريع في مصفوفة
    const projects = [
        {
            title: "Admin Dashboard",
            description: "لوحة تحكم إدارية متكاملة تتيح للمستخدم إدارة المحتوى والمستخدمين"
        },
        {
            title: "E-commerce Platform",
            description: "منصة تجارة إلكترونية متكاملة تتيح للتجار عرض منتجاتهم"
        },
        {
            title: "Social Media App",
            description: "تطبيق تواصل اجتماعي يتيح للمستخدمين التفاعل ومشاركة المحتوى"
        }
    ];

    return (
        <div>
            <h1 className="page-title">مشاريعي</h1>
            {/* استخدام map لعرض جميع المشاريع */}
            {projects.map((project, index) => (
                <Project
                    key={index}
                    title={project.title}
                    description={project.description}
                />
            ))}
        </div>
    );
}

// تحميل التطبيق
const root = ReactDOM.createRoot(document.getElementById('root'));
root.render(<App />);
مميزات هذا الحل:
  • مكون واحد قابل لإعادة الاستخدام
  • سهولة إضافة مشاريع جديدة عن طريق إضافة بيانات للمصفوفة فقط
  • سهولة تعديل شكل جميع المشاريع من مكان واحد
  • كود أنظف وأسهل في الصيانة

كيف يعمل Props؟

  • Props هي طريقة لتمرير البيانات من المكون الأب إلى المكون الابن
  • يمكن تمرير أي نوع من البيانات: نصوص، أرقام، مصفوفات، كائنات
  • المكون الابن يستقبل البيانات كمعامل (parameter) في الدالة
  • البيانات التي يتم تمريرها عبر Props هي للقراءة فقط (read-only)

المرحلة الثالثة: استخدام Props مع المصفوفات

نقوم بتخزين بيانات المشاريع في مصفوفة واستخدام map للتكرار:

function App() {
    const projects = [
        {
            title: "لوحة التحكم",
            description: "لوحة تحكم إدارية..."
        },
        {
            title: "متجر إلكتروني",
            description: "منصة تجارة إلكترونية..."
        }
    ];

    return (
        <div>
            {projects.map((project, index) => (
                <Project
                    key={index}
                    title={project.title}
                    description={project.description}
                />
            ))}
        </div>
    );
}

فوائد استخدام Props:

  • إعادة استخدام المكونات بسهولة
  • تقليل تكرار الكود
  • سهولة الصيانة والتعديل
  • إمكانية إضافة مشاريع جديدة بدون تكرار الكود
